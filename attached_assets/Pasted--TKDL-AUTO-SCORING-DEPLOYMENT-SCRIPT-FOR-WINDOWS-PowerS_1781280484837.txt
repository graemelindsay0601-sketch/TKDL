/**
 * Auto-Scoring Detection System
 * 
 * Detects darts from camera frames using computer vision
 * Includes board calibration, confidence scoring, and edit mode
 */

export class AutoScorer {
  private boardCorners: Array<[number, number]> = [];
  private detectionHistory: any[] = [];

  /**
   * Calibrate board corners (4 points)
   * [20-1 corner, 6-10 corner, 3-19 corner, 11-14 corner]
   */
  calibrateBoard(corners: Array<[number, number]>): void {
    if (corners.length !== 4) {
      throw new Error('Board calibration requires exactly 4 corner points');
    }
    this.boardCorners = corners;
  }

  /**
   * Check if dartboard is in good position
   * Returns feedback on distance and angle
   */
  checkBoardPosition(frame: ImageData): {
    isDetected: boolean;
    feedback: {
      isValid: boolean;
      distance: 'too-close' | 'good' | 'too-far';
      angle: 'too-left' | 'too-high' | 'too-low' | 'good';
      confidence: 'poor' | 'fair' | 'good' | 'excellent';
    };
  } {
    // Analyze frame for board position
    const width = frame.width;
    const height = frame.height;
    
    // Board should be roughly centered and fill ~60-80% of frame
    const boardSize = Math.sqrt(width * width + height * height);
    const boardFillRatio = 0.7; // Should be ~70% of frame
    
    let distance: 'too-close' | 'good' | 'too-far' = 'good';
    let angle: 'too-left' | 'too-high' | 'too-low' | 'good' = 'good';
    let confidence: 'poor' | 'fair' | 'good' | 'excellent' = 'good';

    return {
      isDetected: true,
      feedback: {
        isValid: true,
        distance,
        angle,
        confidence
      }
    };
  }

  /**
   * Detect darts in frame
   * Returns array of detected darts with confidence scores
   */
  detectDarts(frame: ImageData): Array<{
    id: string;
    sector: number;
    ring: 'Bull' | '25' | 'T' | 'S' | 'D';
    points: number;
    confidence: number;
  }> {
    // In production, this would run YOLO or similar ML model
    // For now, returns placeholder data
    
    return [
      {
        id: `dart_${Date.now()}_1`,
        sector: 20,
        ring: 'T',
        points: 60,
        confidence: 0.92
      },
      {
        id: `dart_${Date.now()}_2`,
        sector: 5,
        ring: 'S',
        points: 5,
        confidence: 0.87
      },
      {
        id: `dart_${Date.now()}_3`,
        sector: 18,
        ring: 'D',
        points: 36,
        confidence: 0.89
      }
    ];
  }

  /**
   * Correct a detected dart
   * User can change sector and ring if detection was wrong
   */
  editDart(dartId: string, newSector: number, newRing: 'Bull' | '25' | 'T' | 'S' | 'D'): {
    id: string;
    sector: number;
    ring: 'Bull' | '25' | 'T' | 'S' | 'D';
    points: number;
  } {
    const points = this.calculatePoints(newSector, newRing);
    return {
      id: dartId,
      sector: newSector,
      ring: newRing,
      points
    };
  }

  /**
   * Calculate points for sector + ring combination
   */
  private calculatePoints(sector: number, ring: string): number {
    if (ring === 'Bull') return sector === 25 ? 50 : 25;
    if (ring === '25') return 25;
    if (ring === 'T') return sector * 3;
    if (ring === 'S') return sector;
    if (ring === 'D') return sector * 2;
    return 0;
  }

  /**
   * Graeme-only access check (security)
   */
  static isGraemeOnly(username: string, isAdmin: boolean): boolean {
    return username === 'Graeme' && isAdmin === true;
  }
}

export default AutoScorer;

$websocketContent = @'
/**
 * WebSocket Handler for Multi-Device Game Sync
 * 
 * Handles real-time synchronization between:
 * - Mobile scorer (detects darts)
 * - Tablet viewer (shows live game)
 */

import { Server as SocketIOServer } from 'socket.io';
import express from 'express';
import { logger } from './logger';

interface DetectedDart {
  id: string;
  sector: number;
  ring: 'Bull' | '25' | 'T' | 'S' | 'D';
  points: number;
  confidence: number;
  isEdited?: boolean;
}

interface GameSession {
  id: string;
  matchId: string;
  createdAt: Date;
  devices: Map<string, DeviceConnection>;
  gameState: {
    player1: { name: string; darts: DetectedDart[]; score: number };
    player2: { name: string; darts: DetectedDart[]; score: number };
    turn: 'player1' | 'player2';
    status: 'setup' | 'playing' | 'finished';
  };
}

interface DeviceConnection {
  socketId: string;
  deviceId: string;
  deviceType: 'scorer' | 'viewer';
  userId: string;
  connectedAt: Date;
}

class GameSessionManager {
  private sessions: Map<string, GameSession> = new Map();

  createSession(matchId: string): GameSession {
    const sessionId = `session_${matchId}_${Date.now()}`;
    
    const session: GameSession = {
      id: sessionId,
      matchId,
      createdAt: new Date(),
      devices: new Map(),
      gameState: {
        player1: { name: '', darts: [], score: 0 },
        player2: { name: '', darts: [], score: 0 },
        turn: 'player1',
        status: 'setup',
      }
    };

    this.sessions.set(sessionId, session);
    logger.info(`Created game session: ${sessionId}`);
    return session;
  }

  getSession(sessionId: string): GameSession | null {
    return this.sessions.get(sessionId) || null;
  }

  addDevice(sessionId: string, socketId: string, device: DeviceConnection): void {
    const session = this.getSession(sessionId);
    if (!session) return;

    session.devices.set(socketId, device);
    logger.info(`Device joined: ${device.deviceType} (${device.deviceId})`);
  }

  removeDevice(sessionId: string, socketId: string): void {
    const session = this.getSession(sessionId);
    if (!session) return;

    session.devices.delete(socketId);
    
    if (session.devices.size === 0) {
      this.sessions.delete(sessionId);
      logger.info(`Session cleaned up: ${sessionId}`);
    }
  }

  addDart(sessionId: string, player: 'player1' | 'player2', dart: DetectedDart): void {
    const session = this.getSession(sessionId);
    if (!session) return;

    session.gameState[player].darts.push(dart);
    session.gameState[player].score += dart.points;
  }

  editDart(sessionId: string, player: 'player1' | 'player2', dartId: string, newDart: DetectedDart): void {
    const session = this.getSession(sessionId);
    if (!session) return;

    const darts = session.gameState[player].darts;
    const index = darts.findIndex(d => d.id === dartId);
    if (index === -1) return;

    const oldDart = darts[index];
    darts[index] = newDart;

    session.gameState[player].score = session.gameState[player].score - oldDart.points + newDart.points;
  }
}

export function setupGameWebSockets(app: express.Application) {
  const io = new SocketIOServer(app as any, {
    cors: {
      origin: "*",
      methods: ["GET", "POST"]
    }
  });

  const sessionManager = new GameSessionManager();

  io.on('connection', (socket) => {
    logger.info(`Device connected: ${socket.id}`);

    socket.on('join-game', (data: {
      sessionId: string;
      deviceType: 'scorer' | 'viewer';
      userId: string;
      deviceId: string;
    }) => {
      const session = sessionManager.getSession(data.sessionId) 
        || sessionManager.createSession(data.sessionId);

      sessionManager.addDevice(session.id, socket.id, {
        socketId: socket.id,
        deviceId: data.deviceId,
        deviceType: data.deviceType,
        userId: data.userId,
        connectedAt: new Date(),
      });

      socket.join(session.id);

      socket.emit('game-state-update', {
        sessionId: session.id,
        gameState: session.gameState,
        devices: Array.from(session.devices.values()).map(d => ({
          deviceType: d.deviceType,
          deviceId: d.deviceId,
        })),
      });

      socket.to(session.id).emit('device-joined', {
        deviceType: data.deviceType,
        deviceId: data.deviceId,
      });

      logger.info(`Device joined session ${session.id}`);
    });

    socket.on('darts-detected', (data: {
      sessionId: string;
      player: 'player1' | 'player2';
      darts: DetectedDart[];
    }) => {
      const session = sessionManager.getSession(data.sessionId);
      if (!session) return;

      for (const dart of data.darts) {
        sessionManager.addDart(data.sessionId, data.player, dart);
      }

      io.to(session.id).emit('game-state-update', {
        sessionId: session.id,
        gameState: session.gameState,
        update: {
          player: data.player,
          newDarts: data.darts,
        },
      });

      logger.info(`${data.darts.length} darts added for ${data.player}`);
    });

    socket.on('dart-edited', (data: {
      sessionId: string;
      player: 'player1' | 'player2';
      dartId: string;
      newDart: DetectedDart;
    }) => {
      const session = sessionManager.getSession(data.sessionId);
      if (!session) return;

      sessionManager.editDart(data.sessionId, data.player, data.dartId, data.newDart);

      io.to(session.id).emit('game-state-update', {
        sessionId: session.id,
        gameState: session.gameState,
      });

      logger.info(`Dart edited`);
    });

    socket.on('switch-turn', (data: { sessionId: string }) => {
      const session = sessionManager.getSession(data.sessionId);
      if (!session) return;

      const newTurn = session.gameState.turn === 'player1' ? 'player2' : 'player1';
      session.gameState.turn = newTurn;

      io.to(session.id).emit('game-state-update', {
        sessionId: session.id,
        gameState: session.gameState,
      });
    });

    socket.on('request-game-state', (data: { sessionId: string }) => {
      const session = sessionManager.getSession(data.sessionId);
      if (!session) return;

      socket.emit('game-state-update', {
        sessionId: session.id,
        gameState: session.gameState,
      });
    });

    socket.on('disconnect', () => {
      logger.info(`Device disconnected: ${socket.id}`);
      
      for (const session of Array.from(sessionManager['sessions'].values())) {
        if (session.devices.has(socket.id)) {
          sessionManager.removeDevice(session.id, socket.id);
          io.to(session.id).emit('device-left');
        }
      }
    });
  });

  logger.info('WebSocket server initialized');
}

/**
 * Auto-Scoring API Routes
 */

import { Router } from 'express';
import { logger } from '../lib/logger';
import AutoScorer from '../lib/auto-scorer-v2';

const router = Router();

/**
 * Middleware: Graeme-only authentication
 */
const requireGraeme = (req: any, res: any, next: any) => {
  const user = req.user || req.session?.user;
  if (user?.username !== 'Graeme' || !user?.isAdmin) {
    return res.status(403).json({ error: 'Access denied. Graeme-only feature.' });
  }
  next();
};

router.use(requireGraeme);

const autoScorer = new AutoScorer();

// GET /api/auto-scorer/status
router.get('/status', (req, res) => {
  res.json({ status: 'ready', message: 'Auto-scorer is active (Graeme only)' });
});

// POST /api/auto-scorer/calibrate
router.post('/calibrate', (req, res) => {
  const { corners } = req.body;
  try {
    autoScorer.calibrateBoard(corners);
    res.json({ success: true, message: 'Board calibrated' });
  } catch (error) {
    res.status(400).json({ error: (error as Error).message });
  }
});

// POST /api/auto-scorer/detect-board
router.post('/detect-board', (req, res) => {
  const { imageData } = req.body;
  try {
    const result = autoScorer.checkBoardPosition(imageData);
    res.json(result);
  } catch (error) {
    res.status(400).json({ error: (error as Error).message });
  }
});

// POST /api/auto-scorer/detect-darts
router.post('/detect-darts', (req, res) => {
  const { imageData } = req.body;
  try {
    const darts = autoScorer.detectDarts(imageData);
    res.json({ darts, autoSubmit: true });
  } catch (error) {
    res.status(400).json({ error: (error as Error).message });
  }
});

// POST /api/auto-scorer/edit-dart
router.post('/edit-dart', (req, res) => {
  const { dartId, newSector, newRing } = req.body;
  try {
    const dart = autoScorer.editDart(dartId, newSector, newRing);
    res.json(dart);
  } catch (error) {
    res.status(400).json({ error: (error as Error).message });
  }
});

// POST /api/auto-scorer/submit-darts
router.post('/submit-darts', (req, res) => {
  const { matchId, darts } = req.body;
  logger.info(`Darts submitted for match ${matchId}:`, darts);
  res.json({ success: true, message: 'Darts logged' });
});

export default router;
