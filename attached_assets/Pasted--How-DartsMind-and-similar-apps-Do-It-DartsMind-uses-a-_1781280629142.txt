typescript// artifacts/tkdl/src/lib/dart-detector.ts

import * as tflite from 'react-native-tflite';
import { Camera } from 'react-native-camera-ml-vision';

export class DartDetector {
  private interpreter: any;
  private boardCalibration: BoardCalibrationPoints | null = null;

  async initialize() {
    // Load the TFLite model
    await tflite.loadModel(
      modelName: 'dart_detection.tflite',
      labelName: 'dart_labels.txt',
    );
  }

  async detectDarts(imageFrame: ImageFrame): Promise<DetectedDart[]> {
    // 1. Get dart bounding boxes from YOLO
    const detections = await tflite.runModelOnImage({
      image: imageFrame,
      imageMean: 127.5,
      imageStd: 127.5,
      numResults: 3, // Max 3 darts per frame
      threshold: 0.4, // Confidence threshold
    });

    if (!this.boardCalibration) {
      // First run: calibrate the board
      this.boardCalibration = await this.calibrateBoard(imageFrame);
    }

    // 2. Convert pixel coordinates to board positions
    const darts: DetectedDart[] = detections.map(det => {
      const { x, y } = det;
      const boardPosition = this.pixelToBoard(x, y);
      return {
        boardPosition, // { sector: 20, ring: "T" }
        confidence: det.confidence,
        timestamp: Date.now(),
      };
    });

    return darts;
  }

  private async calibrateBoard(frame: ImageFrame): Promise<BoardCalibrationPoints> {
    // Detect the 4 outer calibration points (usually 20-1, 6-10, 3-19, 11-14 intersections)
    // These are known positions on the board that YOLO can reliably detect
    const calibPoints = await this.detectCalibrationPoints(frame);
    
    // Store transformation matrix for pixel → board conversion
    return calibPoints;
  }

  private pixelToBoard(pixelX: number, pixelY: number): BoardPosition {
    // Use homography transformation to convert pixel coords to board sectors
    const [centerX, centerY] = this.boardCenter;
    const dx = pixelX - centerX;
    const dy = pixelY - centerY;

    const angle = Math.atan2(dy, dx);
    const distance = Math.sqrt(dx * dx + dy * dy);

    // Map to dartboard rings (Bull, 25, T, S, D)
    const ring = this.distanceToRing(distance);
    
    // Map angle to sector (1-20)
    const sector = this.angleToSector(angle);

    return { sector, ring };
  }

  private distanceToRing(distance: number): "Bull" | "25" | "T" | "S" | "D" {
    // Typical dartboard measurements:
    // Bull: 0-10mm from center
    // 25 Ring: 10-20mm
    // Triple Ring: 97-107mm
    // Single: 107-160mm
    // Double Ring: 160-170mm
    
    if (distance < 10) return "Bull";
    if (distance < 20) return "25";
    if (distance > 97 && distance < 107) return "T";
    if (distance > 160 && distance < 170) return "D";
    return "S";
  }

  private angleToSector(angle: number): number {
    // Dartboard sectors go 1-20 in specific order
    // 20 is at top, angles increase clockwise
    const sectors = [20, 1, 18, 4, 13, 6, 10, 15, 2, 17, 3, 19, 7, 16, 8, 11, 14, 9, 12, 5];
    const index = Math.round((angle + Math.PI) / (2 * Math.PI) * 20) % 20;
    return sectors[index];
  }
}

import { DartDetector } from '@/lib/dart-detector';

export function PlayPage() {
  const [detectedDarts, setDetectedDarts] = useState<DetectedDart[]>([]);
  const [score, setScore] = useState(0);
  const cameraRef = useRef(null);
  const detectorRef = useRef(new DartDetector());
  const frameCountRef = useRef(0);

  useEffect(() => {
    detectorRef.current.initialize();
  }, []);

  const handleCameraFrame = async (frame: CameraFrame) => {
    // Only process every 10th frame to reduce load (3 FPS for inference)
    frameCountRef.current++;
    if (frameCountRef.current % 10 !== 0) return;

    const darts = await detectorRef.current.detectDarts(frame);
    
    // Only add if confidence high enough
    const highConfidenceDarts = darts.filter(d => d.confidence > 0.8);
    
    if (highConfidenceDarts.length > 0) {
      setDetectedDarts(prev => [...prev, ...highConfidenceDarts]);
      
      // Auto-submit to server
      highConfidenceDarts.forEach(dart => {
        submitDartToMatch({
          matchId,
          position: dart.boardPosition,
          source: 'AUTO_DETECT',
          confidence: dart.confidence,
        });
      });
    }
  };

  return (
    <div className="play-screen">
      <Camera
        ref={cameraRef}
        onFrame={handleCameraFrame}
        type="back"
        facing="portrait"
      />

      <div className="detected-darts">
        <h3>Detected Darts</h3>
        {detectedDarts.slice(-3).map((dart, i) => (
          <div key={i}>
            {dart.boardPosition.ring}{dart.boardPosition.sector}
            <span className="confidence">
              {(dart.confidence * 100).toFixed(0)}%
            </span>
          </div>

export class DartboardDataCollector {
  private collectionBuffer: CameraFrame[] = [];
  private collectionThreshold = 10; // Every 10 matches

  async onMatchComplete(matchId: string, frames: CameraFrame[]) {
    // Add to collection buffer
    this.collectionBuffer.push(...frames);

    // Every 10 matches, save high-confidence frames to dataset
    if (this.collectionBuffer.length >= this.collectionThreshold) {
      await this.saveDatasetFrames(matchId);
      this.collectionBuffer = [];
    }
  }

  private async saveDatasetFrames(batchId: string) {
    // Save frames with metadata
    const dataset = {
      batchId,
      timestamp: new Date(),
      frames: this.collectionBuffer,
      boardConditions: {
        lighting: 'home',
        boardType: 'standard_bristle',
        dartColor: 'mixed',
      },
    };

    // Save to local storage/database
    await fs.writeFile(
      `/data/dart-training-data/${batchId}.json`,
      JSON.stringify(dataset)
    );

    console.log(`✅ Saved ${this.collectionBuffer.length} frames to dataset`);
  }
}
